Non récusif (plus efficace que le récussif ici)

def dichotomie(L,x):
    a,b=0,len(L)
    while b-a>0:
        m=(a+b)//2
        if x>L[m]:
            a=m+1
        elif x<L[m]:
            b=m-1
        else:
            return(True,m)
    return(False)

def dichotomie_fonction(f,a,b,epsilon):
    while b-a>epsilon:
        m=(a+b)//2
        if f(a)*f(m)>0:
            a=m
        elif f(a)*f(m)<0:
            b=m
        else:
            return(m)
    return((a+b)/2)

def newton(f,fp,x0,epsilon):
    while abs(f(x0))>epsilon: #suffisament proche de 0
        if abs(fp(x0))<1e-10: #pente trop faible = convergence très lente, division par 0...
            break
        else:
            x0=x0-f(x0)/fp(x0)
    return(x0)

#Insérer L[j] dans L[0,j] dejà triée par ordre croissant.
def inserer(L,j):
    k=j
    while k>0 and L[k]<L[k-1]:
        L[k],L[k-1]=L[k-1],L[k]
        k-=1
    return(L)

def fusion(L1,L2):
    i1,i2=0,0
    L=[]

    for _ in range(len(L1)+len(L2)):
        if i1==len(L1):
            L.append(L2[i2])
            i2+=1
        elif i2==len(L2):
            L.append(L1[i1])
            i1+=1
        elif L1[i1]<L2[i2]:
            L.append(L1[i1])
            i1+=1
        else:
            L.append(L2[i2])
            i2+=1
    return(L)

def tri_fusion(L):
    if len(L)<=1:
        return(L)
    else:
        m=len(L)//2
        return(fusion(tri_fusion(L[:m]),tri_fusion(L[m:])))

def tri_rapide(L):
    if len(L)<=1:
        return(L)
    else:
        pivot=L[len(L)//2]
        superieurs=[]
        inferieurs=[]
        egaux=[]

        for x in L:
            if x>pivot:
                superieurs.append(x)
            elif x<pivot:
                inferieurs.append(x)
            else:
                egaux.append(x)
    return(tri_rapide(inferieurs)+egaux+tri_rapide(superieurs))

Intégration numérique¶

Limites de l'intégration numérique :¶


La méthode des trapèzes est plus précise que la méthode des rectangles mais toutes les deux ne fournissent qu'une approximation de l'intégrale.
L'intégration numérique provoque une ''dérive'' (drif) dans le temps qui est d'autant plus grande que le bruit de mesure est grand: il faut filtrer les mesures.

Dérivation numérique¶

Limites de la dérivation numérique :¶


Les méthodes de dérivation numérique classiques provoquent des effets de bords : au début et à la fin du calcul les valeurs de la dérivée ne sont pas réalistes.
La dérivation numérique amplifie les bruits de mesure, il faut donc avoir un signal filtré (ou lissé) pour obtenir des valeurs de dérivée réalistes.

In [24]:
def derivee_centree(Y,Tps):
    Yp=[0.]*len(Y)
    for k in range(1,len(Y)-1):
        Yp[k]=(Y[k+1]-Y[k-1])/(Tps[k+1]-Tps[k-1])
    return(Yp)

#Fonction finale pivot de gauss avec recherhe d'un pivot non nul
def pivot_gauss(M):
    n=np.shape(M)[0]

    choix_pivot(M) #Option : recherche de pivot non nul

    #Descente
    for k in range(n):
        #Pivot = 1
        I[k,:]=I[k,:]/M[k,k]
        M[k,:]=M[k,:]/M[k,k]

        for i in range(k+1,n):
            # que des 0 en dessous du pivot
            I[i,:]=I[i,:]-I[k,:]*M[i,k]
            M[i,:]=M[i,:]-M[k,:]*M[i,k]

    #Remontée
    for k in range(n-1,0,-1): 
        for i in range(k-1,-1,-1): #erreur dans le QCM : jusqu'à 0 exclu
            I[i,:]=I[i,:]-I[k,:]*M[i,k]
            M[i,:]=M[i,:]-M[k,:]*M[i,k]
    return(I)


##Option : recherche de pivot non nul

#Echanger les lignes Li et Lj
def echange(M,i,j):
    L=M[j,:].copy()
    M[j,:]=M[i,:]
    M[i,:]=L

#Recherche d'un pivot non nul (colonne Cj) sur les lignes en dessous de Lj
def recherche_pivot(M,j):
    nbre_lignes=np.shape(M)[0]
    for i in range(j,nbre_lignes):
        if M[i,j]!=0:
            return(i)

#Choix d'un pivot non nul
def choix_pivot(M):
    nbre_col=np.shape(M)[1]
    for k in range(nbre_col-1):
        if M[k,k]==0:
            i_ligne_pivot=recherche_pivot(M,k)
            echange(I,k,i_ligne_pivot)
            echange(M,k,i_ligne_pivot)
